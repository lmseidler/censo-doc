16def configure(rcpath: str = None, create_new: bool = False):
17 """
18 Configures the application based on the provided configuration file path.
19 If no configuration file path is provided, it searches for the default configuration file.
20 If no configuration file is found, it creates a new one with default settings.
21 """
22 # Try to find the .censo2rc in the user's home directory
23 # if no configuration file path is provided
24 if rcpath is None:
25 censorc_path = find_rcfile()
26 else:
27 if not os.path.isfile(rcpath) and not create_new:
28 raise FileNotFoundError(
29 f"No configuration file found at {rcpath}.")
30 else:
31 censorc_path = rcpath
32
33 # Set up the DFAHelper
34 DfaHelper.set_dfa_dict(os.path.join(
35 ASSETS_PATH, "censo_dfa_settings.json"))
36
37 # Set up the SolventHelper
38 SolventHelper.set_solvent_dict(os.path.join(
39 ASSETS_PATH, "censo_solvents_db.json"))
40
41 # map the part names to their respective classes
42 # NOTE: the DFAHelper and the databases should be setup before the parts are imported,
43 # otherwise there will be errors in the CensoPart._options
44 from .part import CensoPart
45 from .ensembleopt import Prescreening, Screening, Optimization, Refinement
46 from .properties import NMR
47
48 global parts
49 parts = {
50 "prescreening": Prescreening,
51 "screening": Screening,
52 "optimization": Optimization,
53 "refinement": Refinement,
54 "nmr": NMR,
55 }
56
57 # If no configuration file was found above, set the rcflag to False
58 global homerc
59 if censorc_path is None:
60 homerc = False
61 return
62 # if explicitely told to create a new configuration file, do so
63 elif create_new:
64 censorc_path = os.path.join(rcpath, "censo2rc_NEW")
65 write_rcfile(censorc_path)
66 # Otherwise, read the configuration file and configure the parts with the settings from it
67 else:
68 homerc = True
69 settings_dict = read_rcfile(censorc_path)
70
71 # first set general settings
72 CensoPart.set_general_settings(settings_dict["general"])
73
74 # set settings for each part
75 for section, settings in settings_dict.items():
76 if section in parts.keys():
77 parts[section].set_settings(settings)
78 # NOTE: if section is not in the parts names, it will be ignored
79
80 # Update the paths for the processors
81 paths = read_rcfile(censorc_path)["paths"]
82 QmProc._paths.update(paths)
83
84 # create user assets folder if it does not exist
85 if not os.path.isdir(USER_ASSETS_PATH):
86 os.mkdir(USER_ASSETS_PATH)

89def read_rcfile(path: str) -> dict[str, dict[str, any]]:
90 """
91 Read from config data from file located at 'path'
92 """
93 # read config file
94 parser: configparser.ConfigParser = configparser.ConfigParser()
95 with open(path, "r") as file:
96 parser.read_file(file)
97
98 returndict = {section: dict(parser[section])
99 for section in parser.sections()}
100 return returndict

103def write_rcfile(path: str) -> None:
104 """
105 Write new configuration file with default settings into file at 'path'.
106 Also reads program paths from preexisting configuration file or tries to
107 determine the paths automatically.
108
109 Args:
110 path (str): Path to the new configuration file.
111
112 Returns:
113 None
114 """
115 # what to do if there is an existing configuration file
116 external_paths = None
117 if os.path.isfile(path):
118 print(
119 f"An existing configuration file has been found at {path}.\n",
120 f"Renaming existing file to {CENSORCNAME}_OLD.\n",
121 )
122 # Read program paths from the existing configuration file
123 print("Reading program paths from existing configuration file ...")
124 external_paths = read_program_paths(path)
125
126 # Rename existing file
127 os.rename(path, f"{path}_OLD")
128
129 with open(path, "w", newline=None) as rcfile:
130 parser = configparser.ConfigParser()
131
132 # collect all default settings from parts and feed them into the parser
133 global parts
134 from .part import CensoPart
135
136 parts["general"] = CensoPart
137 parser.read_dict(
138 {
139 partname: {
140 settingname: setting["default"]
141 for settingname, setting in part.get_options().items()
142 }
143 for partname, part in parts.items()
144 }
145 )
146
147 # Try to get paths from 'which'
148 if external_paths is None:
149 print("Trying to determine program paths automatically ...")
150 external_paths = find_program_paths()
151
152 parser["paths"] = external_paths
153
154 print(f"Writing new configuration file to {path} ...")
155 parser.write(rcfile)
156
157 print(
158 f"\nA new configuration file was written into {path}.\n"
159 "You should adjust the settings to your needs and set the program paths.\n"
160 "Right now the settings are at their default values.\n"
161 )
162
163 if CENSORCNAME not in path:
164 print(
165 f"Additionally make sure that the file name is '{CENSORCNAME}'.\n"
166 f"Currently it is '{os.path.split(path)[-1]}'.\n"
167 )

170def read_program_paths(path: str) -> dict[str, str] | None:
171 """
172 Read program paths from the configuration file at 'path'
173 """
174 with open(path, "r") as inp:
175 parser = configparser.ConfigParser()
176 parser.read_file(inp)
177
178 try:
179 return dict(parser["paths"])
180 except KeyError:
181 print(f"WARNING: No paths found in {path}")
182 return None

185def find_program_paths() -> dict[str, str]:
186 """
187 Try to determine program paths automatically
188 """
189 # TODO - for now only the most important ones are implemented
190 mapping = {
191 "orcapath": "orca",
192 "xtbpath": "xtb",
193 "crestpath": "crest",
194 "cosmorssetup": None,
195 "dbpath": None,
196 "cosmothermversion": None,
197 "mpshiftpath": None,
198 "escfpath": None,
199 }
200 paths = {}
201
202 for pathname, program in mapping.items():
203 if program is not None:
204 path = shutil.which(program)
205 else:
206 path = None
207
208 if path is not None:
209 paths[pathname] = path
210 else:
211 paths[pathname] = ""
212
213 # if orca was found try to determine orca version from the path (kinda hacky)
214 if paths["orcapath"] != "":
215 try:
216 paths["orcaversion"] = (
217 paths["orcapath"].split(os.sep)[-2][5:10].replace("_", ".")
218 )
219 except Exception:
220 paths["orcaversion"] = ""
221
222 return paths
