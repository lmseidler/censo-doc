56 def tocoord(self) -> list[str]:
57 """
58 method to convert the internal cartesian coordinates (self.xyz) to coord file format (for tm or xtb)
59 """
60 coord = ["$coord\n"]
61 for atom in self.xyz:
62 coord.append(
63 reduce(
64 lambda x, y: f"{x} {y}",
65 list(map(lambda x: float(x) / BOHR2ANG,
66 atom["xyz"])) + [f"{atom['element']}\n"]

68 )

70 coord.append("$end\n")

74 def fromcoord(self, path: str) -> None:
75 """
76 method to convert the content of a coord file to cartesian coordinates for the 'xyz' attribute
77 """
78 with open(path, "r") as file:
79 lines = file.readlines()
80
81 self.xyz = []
82 for line in lines:
83 if not line.startswith("$"):
84 coords = line.split()
85 element = coords[-1]
86 cartesian_coords = [float(x) * BOHR2ANG for x in coords[:-1]]
87 self.xyz.append({"element": element, "xyz": cartesian_coords})
88 elif line.startswith("$end"):
