27 Manages parallel execution of external program calls. Sets cores used per job, checks requirements,
28 can copy MO-files, and retry failed jobs.
29
30 Args:
31 conformers (list[MoleculeData]): List of conformers for which jobs will be created and executed.
32 workdir (str): Working directory.
33 prog (str): Name of the program to be used.
34 copy_mo (bool, optional): Whether to copy the MO-files from the previous calculation.
35 retry_failed (bool, optional): Whether to retry failed jobs.
36 balance (bool, optional): Whether to balance the number of cores used per job.
37 maxcores (int, optional): Maximum number of cores to be used.
38 omp (int, optional): Number of cores to be used per job.
39
40 Returns:
41 dict[int, any]: Dictionary of results. Key: the id of the conformer.
42 """
43

45 try:
46 assert len(conformers) > 0
47 except AssertionError as e:
48 raise e("No jobs to compute!")

51 jobs = prepare_jobs(conformers, prepinfo, jobtype)

55 prog,
56 workdir,

71 try:
72 job.mo_guess = next(
73 c for c in conformers if c.geom.id == job.conf.id

76 pass

123def reduce_cores(
124 free_cores: multiprocessing.Value, omp: int, enough_cores: multiprocessing.Condition
125):
126 # acquire lock on the condition and wait until enough cores are available
127 with enough_cores:
128 enough_cores.wait_for(lambda: free_cores.value >= omp)
129 free_cores.value -= omp

131 f"Free cores decreased {free_cores.value + omp} -> {free_cores.value}."

152def dqp(jobs: list[ParallelJob], processor: QmProc) -> list[ParallelJob]:
153 """
154 D ynamic Q ueue P rocessing
155 """
156
157 global ncores
158
159 with multiprocessing.Manager() as manager:
160 # execute calculations for given list of conformers
161 with ProcessPoolExecutor(
162 max_workers=ncores // min(job.omp for job in jobs)
163 ) as executor:
164 # make sure that the executor exits gracefully on termination
165 # TODO - is using wait=False a good option here?
166 # should be fine since workers will kill programs with SIGTERM
167 # wait=True leads to the workers waiting for their current task to be finished before terminating
168 # Register the signal handler
169 signal.signal(
170 signal.SIGTERM,
171 lambda signum, frame: handle_sigterm(signum, frame, executor),
172 )
173
174 # define shared variables that can be safely asynchronously accessed
175 free_cores = manager.Value(int, ncores)
176 enough_cores = manager.Condition()
177
178 # sort the jobs by the number of cores used
179 # (the first item will be the one with the lowest number of cores)
180 jobs.sort(key=lambda x: x.omp)
181
182 tasks = []
183 for i in range(len(jobs)):
184 # TODO - something to readjust omp based on expected time to finish and the timings of other jobs
185 # try to reduce the number of cores by job.omp, if there are not enough cores available we wait
186 reduce_cores(free_cores, jobs[i].omp, enough_cores)
187
188 try:
189 # submit the job
190 tasks.append(executor.submit(processor.run, jobs[i]))
191 # NOTE: explanation of the lambda: the first argument passed to the done_callback is always the future
192 # itself, it is not assigned (_), the second parameter is the number of openmp threads of the job (i.e.
193 # job.omp) if this is not specified like this (omp=jobs[i].omp) the done_callback will instead use the
194 # omp of the current item in the for-iterator (e.g. the submitted job has omp=4, but the current jobs[i]
195 # has omp=7, so the callback would use 7 instead of 4)
196 tasks[-1].add_done_callback(
197 lambda _, omp=jobs[i].omp: increase_cores(
198 free_cores, omp, enough_cores
199 )
200 )
201 except RuntimeError:
202 # Makes this exit gracefully in case that the main process is killed
203 return None
204
205 # wait for all jobs to finish and collect results
206 results = [task.result() for task in as_completed(tasks)]
207
208 return results

211def set_omp_chunking(jobs: list[ParallelJob]) -> None:
212 """
213 Determines and sets the number of cores that are supposed to be used for every job.
214 This method is efficient if it can be assumed that the jobs take roughly the same amount of time each.
215 Each job shouldn't use less than OMPMIN cores.
216 """
217 global ncores # Access the global variable ncores
218
219 # Get the total number of jobs
220 jobs_left, tot_jobs = len(jobs), len(jobs)
221
222 # Calculate the maximum and minimum number of processes (number of jobs that can be executed simultaneously)
223 maxprocs = ncores // OMPMIN # Calculate the maximum number of processes
224 # Calculate the minimum number of processes
225 minprocs = max(1, ncores // OMPMAX)
226
227 # Loop until all jobs are distributed
228 while jobs_left > 0:
229 if jobs_left >= maxprocs:
230 p = maxprocs # Set the number of processes to the maximum if there are enough jobs left
231 elif jobs_left < minprocs:
232 p = minprocs # Set the number of processes to the minimum if there are less jobs left than minprocs
233 else:
234 # Find the largest number of processes that evenly divides the remaining jobs
235 p = max(
236 [
237 j
238 for j in range(minprocs, maxprocs)
239 if ncores % j == 0 and j <= jobs_left
240 ]
241 )
242
243 # Set the number of cores for each job for as many jobs as possible before moving onto the next omp value
244 while jobs_left - p >= 0:
245 for job in jobs[tot_jobs - jobs_left: tot_jobs - jobs_left + p]:
246 job.omp = ncores // p # Set the number of cores for each job
247 jobs_left -= p # Decrement the number of remaining jobs

250def retry_failed_jobs(jobs: list[ParallelJob], processor: QmProc) -> tuple[list[int], list[int]]:
251 """
252 Tries to recover failed jobs.
253
254 Args:
255 jobs (list[ParallelJob]): List of jobs.
256 processor (QmProc): Processor object.
257
258 Returns:
259 tuple[list[int], list[int]]: List of indices of jobs that should be retried, list of ids of conformers
260 that could not be recovered.
261 """
262 # determine failed jobs
263 logger.debug("Retrying failed jobs...")
264 failed_jobs = [
265 i
266 for i, job in enumerate(jobs)
267 if any(not job.meta[jt]["success"] for jt in job.jobtype)
268 ]
269
270 if len(failed_jobs) != 0:
271 # create a new list of failed jobs that should be restarted with special flags
272 # contains jobs that should be retried (depends if the error can be handled or not)
273 retry = []
274
275 # determine flags for jobs based on error messages
276 for failed_job in failed_jobs:
277 # list of jobtypes that should be removed from the jobtype list
278 jtremove = []
279 for jt in jobs[failed_job].jobtype:
280 # for now only sp and gsolv calculations are caught
281 if not jobs[failed_job].meta[jt]["success"] and jt in [
282 "sp",
283 "gsolv",
284 ]:
285 if jobs[failed_job].meta[jt]["error"] == "SCF not converged":
286 retry.append(failed_job)
287 jobs[failed_job].flags[jt] = "scf_not_converged"
288 # store all successful jobtypes
289 elif jobs[failed_job].meta[jt]["success"]:
290 jtremove.append(jt)
291
292 # remove all successful jobs from jobtype to avoid re-execution
293 for jt in jtremove:
294 jobs[failed_job].jobtype.remove(jt)
295
296 # execute jobs that should be retried
297 logger.info(
298 f"Failed jobs: {len(failed_jobs)}. Restarting {len(retry)} jobs."
299 )
300
301 if len(retry) > 0:
302 set_omp_chunking([jobs[i] for i in retry])
303 for i, job in zip([i for i in retry], dqp([jobs[i] for i in retry], processor)):
304 jobs[i] = job
305
306 # any jobs that still failed will lead to the conformer being marked as unrecoverable
307 failed_confs = []
308 for job in jobs:
309 if not all(job.meta[jt]["success"] for jt in job.jobtype):
310 logger.warning(
311 f"{job.conf.name} job recovery failed."
312 )
313 failed_confs.append(job.conf.id)
314 else:
315 retry = []
316 failed_confs = []
317 logger.info("All jobs executed successfully.")
318
319 return retry, failed_confs
