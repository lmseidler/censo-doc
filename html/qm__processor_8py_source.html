101 def run(self, job: ParallelJob) -> ParallelJob:
102 """
103 Run methods depending on jobtype.
104 DO NOT OVERRIDE OR OVERLOAD! this will break e.g. censo.parallel.execute
105
106 Args:
107 job (ParallelJob): job to run
108
109 Returns:
110 job (ParallelJob): job with results
111 """
112 logger.debug(
113 f"{f'worker{os.getpid()}:':{WARNLEN}}Running on {job.omp} cores.")
114 # jobtype is basically an ordered (!!!) (important e.g. if sp is required before the next step)
115 # list containing the types of computations to do
116 if not all(t in self._jobtypes.keys() for t in job.jobtype):
117 raise RuntimeError(
118 f"At least one jobtype of {self._jobtypes} is not available for {self.__class__.__name__}.\nAvailable "
119 + f"jobtypes are: {self._jobtypes.keys()}"
120 )
121
122 # run all the computations
123 for j in job.jobtype:
124 # Create jobdir
125 jobdir = self._create_jobdir(job.conf.name, j)
126
127 # Time execution
128 start = perf_counter()
129
130 logger.info(
131 f"{f'worker{os.getpid()}:':{WARNLEN}}Running {j} calculation in {jobdir}."
132 )
133 print(f"Running {j} calculation for {job.conf.name}.")
134 job.results[j], job.meta[j] = self._jobtypes[j](job, jobdir)
135
136 # Copy mo path if possible to be used for further calculations
137 # (processors grab the mo path from the 'mo_path' key that is always present in the meta dict)
138 if job.meta[j].get("mo_path", None) is not None:
139 job.meta["mo_path"] = job.meta[j]["mo_path"]
140
141 end = perf_counter()
142
143 job.meta[j]["time"] = end - start
144
145 # if a calculation failed all following calculations will not be executed
146 if not job.meta[j]["success"]:
147 for j2 in job.jobtype[job.jobtype.index(j) + 1:]:
148 job.meta[j2]["success"] = False
149 job.meta[j2]["error"] = "Previous calculation failed"
150 break
151
152 job.meta["total_time"] = sum(
153 job.meta[j]["time"]
154 for j in job.jobtype
155 if job.meta[j]["error"] != "Previous calculation failed"
156 )
157
158 # returns modified job object with result dict e.g.: {"sp": ..., "gsolv": ..., etc.}
159 return job

161 @staticmethod
162 def _make_call(call: list, outputpath: str, jobdir: str) -> int:
163 """
164 Make a call to an external program and write output into outputfile.
165
166 Args:
167 call (list): list containing the call to the external program
168 outputpath (str): path to the outputfile
169 jobdir (str): path to the jobdir
170
171 Returns:
172 returncode (int): returncode of the external program
173 """
174 # call external program and write output into outputfile
175 with open(outputpath, "w", newline=None) as outputfile:
176 logger.debug(
177 f"{f'worker{os.getpid()}:':{WARNLEN}}Running {call}...")
178
179 # create subprocess for external program
180 sub = subprocess.Popen(
181 call,
182 shell=False,
183 stdin=None,
184 stderr=subprocess.STDOUT,
185 universal_newlines=False,
186 cwd=jobdir,
187 stdout=outputfile,
188 env=ENVIRON,
189 )
190
191 logger.debug(
192 f"{f'worker{os.getpid()}:':{WARNLEN}}Started (PID: {sub.pid})."
193 )
194
195 # make sure to send SIGTERM to subprocess if program is quit
196 signal.signal(
197 signal.SIGTERM, lambda signum, frame: handle_sigterm(
198 signum, frame, sub)
199 )
200
201 # wait for process to finish
202 returncode = sub.wait()
203
204 logger.debug(f"{f'worker{os.getpid()}:':{WARNLEN}}Done.")
205
206 return returncode

227 """Get rotational symmetry number from Schoenfließ symbol"""
228 if sym is None:
229 sym = "c1"
230 if linear is None:
231 linear = False
232 symnum = 1
233 if linear and "c" in sym.lower()[0]:
234 symnum = 1
235 return symnum
236 elif linear and "d" in sym.lower()[0]:
237 symnum = 2
238 return symnum
239 for key in rot_sym_num.keys():
240 if key in sym.lower():

359 def _xtb_gsolv(self, job: ParallelJob, jobdir: str) -> tuple[dict[str, float | None], dict[str, any]]:
360 """
361 Calculate additive GBSA or ALPB solvation using GFNn-xTB or GFN-FF.
362
363 Args:
364 job (ParallelJob): job to run
365 jobdir (str): path to the jobdir
366
367 Returns:
368 result (dict[str, any]): result of the gsolv calculation
369
370 result = {
371 "gsolv": None,
372 "energy_xtb_gas": None,
373 "energy_xtb_solv": None,
374 }
375 """
376 # what is returned in the end
377 result = {
378 "gsolv": None,
379 "energy_xtb_gas": None,
380 "energy_xtb_solv": None,
381 }
382
383 meta = {
384 "success": None,
385 "error": None,
386 }
387
388 # run gas-phase GFN single-point
389 spres, spmeta = self._xtb_sp_xtb_sp(job, jobdir, filename="gas", no_solv=True)
390 if spmeta["success"]:
391 result["energy_xtb_gas"] = spres["energy"]
392 else:
393 meta["success"] = False
394 meta["error"] = "what went wrong in xtb_gsolv"
395 return result, meta
396
397 # run single-point in solution:
398 # ''reference'' corresponds to 1\;bar of ideal gas and 1\;mol/L of liquid
399 # solution at infinite dilution,
400 spres, spmeta = self._xtb_sp_xtb_sp(job, jobdir, filename="solv")
401 if spmeta["success"]:
402 result["energy_xtb_solv"] = spres["energy"]
403 else:
404 meta["success"] = False
405 meta["error"] = "what went wrong in xtb_gsolv"
406 return result, meta
407
408 # only reached if both gas-phase and solvated sp succeeded
409 result["gsolv"] = result["energy_xtb_solv"] - result["energy_xtb_gas"]
410 meta["success"] = True
411
412 return result, meta
